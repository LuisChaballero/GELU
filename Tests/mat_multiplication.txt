
program test0 ;

var int m, n, res;
var int mat[2,2], mat2[2,2];

func void multiplicacionMatrices() {
   var int mat[2,2], mat2[2,2];
   mat[0,0] = 4;
   mat[0,1] = -2;
   mat[1,0] = 3;
   mat[1,1] = 1;
   mat2[0,0] = 2;
   mat2[0,1] = -3;
   mat2[1,0] = 5;
   mat2[1,1] = 0;
   for m = 0 until m > 1 {
       for n = 0 until n > 1 {
           res = (mat[m,n] * mat2[m,n]);
           # Multiplicacion de matrices UWU
           print(mat[m,n], " * ", mat2[m,n], " = ", res);
           print() 
       }
       print("\n");
   }
  
}

main() { 
    
    multiplicacionMatrices();
}
